void conn_msg_from_pins(Pin* Pin1, uint8_t Pin1_num, Pin* Pin2, uint8_t Pin2_num, char* conn_msg, uint8_t* conn_msg_len) {
  if (Pin1 == NULL) {
    sprintf(conn_msg, "main.c: Code Error: Pin1 cannot be NULL.");
    return;
  }

  if (Pin2 == NULL)
    sprintf(conn_msg, "(%u) -> X\n", (Pin1_num+1)); 
  else
    sprintf(conn_msg, "(%u) -> (%u)\n", (Pin1_num+1), (Pin2_num+1)); 
  *conn_msg_len = (uint8_t)strlen(conn_msg);
  return;
}

                conn_msg_from_pins(&Output_Pins[i], i, &Input_Pins[k], k, conn_msg, &conn_msg_len);
                HAL_UART_Transmit(&huart3, (uint8_t*)conn_msg, conn_msg_len, HAL_MAX_DELAY);
              }

              }
